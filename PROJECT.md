## Interface PГєblica (Headergates)

```cpp
// MarkdownWriter.h
class MarkdownWriter {
public:
    struct Options {
        std::string format = "auto";     // auto, table, list, code
        char flattenSeparator = '\0';     // se setado, flatten objetos
        int maxDepth = -1;                // -1 = ilimitado
        bool showNull = true;
        bool showTypes = false;
        std::string title = "";
        int previewLines = -1;            // -1 = todas
    };
    
    std::string convert(const json& data, const Options& opts);
};

// Converter.h
class Converter {
public:
    virtual ~Converter() = default;
    virtual bool canHandle(const std::string& filename) const = 0;
    virtual json parse(const std::string& content) = 0;
    virtual std::string name() const = 0;
};

// JsonConverter.h
class JsonConverter : public Converter {
    bool canHandle(const std::string& filename) const override; // .json
    json parse(const std::string& content) override;
    std::string name() const override { return "JSON"; }
};
```
